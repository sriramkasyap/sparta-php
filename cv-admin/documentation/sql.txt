

CREATE TABLE `cv_cms`.`site_pages` ( `page_id` INT(10) NOT NULL AUTO_INCREMENT , `page_title` VARCHAR(255) NOT NULL , `page_url` TEXT NOT NULL , `page_heading` VARCHAR(255) NOT NULL , `page_type` ENUM('home', 'project_page') NOT NULL , PRIMARY KEY (`page_id`), UNIQUE (`page_url`)) ENGINE = InnoDB;
